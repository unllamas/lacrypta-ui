## Usage

<br />

export function PreviewDrawer({ children }) {
  const { isOpen, onOpen, onClose } = useDisclosure()

  return (
    <Flex align='center' justify='center' style={{ minHeight: '668px' }}>
      <div>
        <Button onClick={onOpen}>Open Drawer</Button>
      </div>
      <Drawer isOpen={isOpen} onClose={onClose}>
        <Input label="Email" placeholder="you@example.com" />
        <Divider y="sm" />
        <Flex gap="sm">
          <Button isBlock variant="bezeledGray" onClick={onClose}>
            Cancelar
          </Button>
          <Button isBlock>Guardar</Button>
        </Flex>
      </Drawer>
    </Flex>
  )
}

<Card variant="outline" spacing={4}>

</Card>
